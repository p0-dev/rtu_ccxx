# ****************************************************************************************
# CMAKE SCRIPT FILE

# ****************************************************************************************
# @location     | PJT_ROOT_DIR_PATH
# ****************************************************************************************
# @brief        | CMake main scrip file
# ****************************************************************************************
# @desc         | Include
#               |   + Target list
#               |   + CMake general settings
#               |   + Sub-modules
# ****************************************************************************************


# ========================================================================================
# Target List
# ----------------------------------------------------------------------------------------
# [Name]            | [Location]    | [Description]
# ----------------------------------------------------------------------------------------
# lib_main          | src/main      | main execution of the project
# ----------------------------------------------------------------------------------------
# lib_testlib1      | src/testlib1  | test library 1
# ----------------------------------------------------------------------------------------
# lib_testlib2      | src/testlib2  | test library 2
# ----------------------------------------------------------------------------------------

# ========================================================================================
# Pre Processing
# ----------------------------------------------------------------------------------------
cmake_minimum_required (VERSION 3.29)
project (${PJT_NAME})

# ========================================================================================
# General Configuration
# ----------------------------------------------------------------------------------------
set (CMAKE_EXPORT_COMPILE_COMMANDS 1)

# ========================================================================================
# Module
# ----------------------------------------------------------------------------------------
add_subdirectory (${CMAKE_SOURCE_DIR}/src/main)
add_subdirectory (${CMAKE_SOURCE_DIR}/src/testlib1)
add_subdirectory (${CMAKE_SOURCE_DIR}/src/testlib2)

# ========================================================================================
# Final Target
# ----------------------------------------------------------------------------------------
add_executable (${PROJECT_NAME})

# ========================================================================================
# Final Target Linking Library
# ----------------------------------------------------------------------------------------
target_link_libraries (
  ${PROJECT_NAME}
  PRIVATE
  lib_main
)

# ========================================================================================
# Final Target Installation
# ----------------------------------------------------------------------------------------
install (
  TARGETS ${PROJECT_NAME} 
  DESTINATION .
)


# ****************************************************************************************
# THE END
